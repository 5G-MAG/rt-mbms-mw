75 void MBMS_RT::RestHandler::get(http_request message) {
76  spdlog::debug("Received GET request {}", message.to_string() );
77  auto uri = message.relative_uri();
78  auto paths = uri::split_path(uri::decode(message.relative_uri().path()));
79  if (_require_bearer_token &&
80  (message.headers()["Authorization"] != "Bearer " + _api_key)) {
81  message.reply(status_codes::Unauthorized);
82  return;
83  }
84 
85  if (paths.empty()) {
86  message.reply(status_codes::NotFound);
87  } else {
88  if (paths[0] == _api_path) {
89  if (paths[1] == "files") {
90  std::vector<value> files;
91  for (const auto& [tmgi, service] : _services) {
92  for (const auto& file : service->fileList()) {
93  value f;
94  f["tmgi"] = value(tmgi);
95  f["access_count"] = value(file->access_count());
96  f["location"] = value(file->meta().content_location);
97  f["content_length"] = value(file->meta().content_length);
98  f["received_at"] = value(file->received_at());
99  f["age"] = value(time(nullptr) - file->received_at());
100  files.push_back(f);
101  }
102  }
103  message.reply(status_codes::OK, value::array(files));
104  } else if (paths[1] == "services") {
105  std::vector<value> services;
106  for (const auto& [tmgi, service] : _services) {
107  if (!service->bootstrapped()) continue;
108  value s;
109  s["service_tmgi"] = value(tmgi);
110  s["service_name"] = value(service->serviceName());
111  s["service_description"] = value(service->serviceDescription());
112  s["sdp"] = value(service->sdp());
113  s["m3u"] = value(service->m3u());
114  s["stream_tmgi"] = value(service->streamTmgi());
115  s["stream_type"] = value(service->streamType());
116  s["stream_mcast"] = value(service->streamMcast());
117  services.push_back(s);
118  }
119  message.reply(status_codes::OK, value::array(services));
120  }
121  } else if (paths[0] == "f") {
122  for (const auto& [tmgi, service] : _services) {
123  if (tmgi == paths[1]) {
124  std::vector<std::string> location(&paths[2],&paths[paths.size()]);
125  auto path = boost::algorithm::join(location, "/");
126  spdlog::debug("searching for location {}", path);
127  for (const auto& file : service->fileList()) {
128  auto file_loc = file->meta().content_location;
129  file_loc = file_loc.substr(0, file_loc.find('?'));
130  spdlog::debug("checking {}", file_loc );
131  bool flute_ffmpeg_enabled = false;
132  _cfg.lookupValue("mw.flute_ffmpeg.enabled", flute_ffmpeg_enabled);
133  std::string target_path = flute_ffmpeg_enabled ? "/" + path : path;
134  if (file_loc == target_path) {
135  spdlog::debug("found!");
136  file->log_access();
137  auto instream = Concurrency::streams::rawptr_stream<uint8_t>::open_istream((uint8_t*)file->buffer(), file->meta().content_length);
138  message.reply(status_codes::OK, instream);
139 
140  }
141  }
142  }
143  }
144  message.reply(status_codes::NotFound);
145  } else {
146  message.reply(status_codes::NotFound);
147  }
148  }
149 }
